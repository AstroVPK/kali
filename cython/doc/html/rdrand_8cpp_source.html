91 /*
92  * The Intel compiler intrinsic for generating a 64-bit rand on a 32-bit
93  * system maps to two 32-bit RDRAND instructions. Because of the way
94  * the way the DRNG is implemented you can do this up to a 128-bit value
95  * (for crypto purposes) before you no longer have multiplicative
96  * prediction resistance.
97  *
98  * Note that this isn't very efficient. If you need 64-bit values
99  * you should really be on a 64-bit system.
100  */

104 int _rdrand64_step (uint64_t *x)
105 {
106  uint32_t xlow, xhigh;
107  int rv;
108 
109  if ( (rv= _rdrand32_step(&xlow)) != RDRAND_SUCCESS ) return rv;
110  if ( (rv= _rdrand32_step(&xhigh)) != RDRAND_SUCCESS ) return rv;
111 
112  *x= (uint64_t) xlow | ((uint64_t)xhigh<<32);
113 
114  return RDRAND_SUCCESS;
115 }

129 int RdRand_cpuid()
130 {
131  int info[4] = {-1, -1, -1, -1};
132 
133  /* Are we on an Intel processor? */
134 
135  __cpuid(info, 0);
136 
137  if ( memcmp((void *) &info[1], (void *) "Genu", 4) != 0 ||
138  memcmp((void *) &info[3], (void *) "ineI", 4) != 0 ||
139  memcmp((void *) &info[2], (void *) "ntel", 4) != 0 ) {
140 
141  return 0;
142  }
143 
144  /* Do we have RDRAND? */
145 
146  __cpuid(info, /*feature bits*/1);
147 
148  int ecx = info[2];
149  if ((ecx & RDRAND_MASK) == RDRAND_MASK)
150  return 1;
151  else
152  return 0;
153 }

307 int rdrand_get_bytes(unsigned int n, unsigned char *dest)
308 {
309  unsigned char *start;
310  unsigned char *residualstart;
311  _wordlen_t *blockstart;
312  _wordlen_t i, temprand;
313  unsigned int count;
314  unsigned int residual;
315  unsigned int startlen;
316  unsigned int length;
317  int success;
318 
319  /* Compute the address of the first 32- or 64- bit aligned block in the destination buffer, depending on whether we are in 32- or 64-bit mode */
320  start = dest;
321  if (((_wordlen_t) start % (_wordlen_t) sizeof(_wordlen_t)) == 0)
322  {
323  blockstart = (_wordlen_t *)start;
324  count = n;
325  startlen = 0;
326  }
327  else
328  {
329  blockstart = (_wordlen_t *)(((_wordlen_t)start & ~(_wordlen_t) (sizeof(_wordlen_t)-1) )+(_wordlen_t)sizeof(_wordlen_t));
330  count = n - (sizeof(_wordlen_t) - (unsigned int)((_wordlen_t)start % sizeof(_wordlen_t)));
331  startlen = (unsigned int)((_wordlen_t)blockstart - (_wordlen_t)start);
332  }
333 
334  /* Compute the number of 32- or 64- bit blocks and the remaining number of bytes */
335  residual = count % sizeof(_wordlen_t);
336  length = count/sizeof(_wordlen_t);
337  if (residual != 0)
338  {
339  residualstart = (unsigned char *)(blockstart + length);
340  }
341 
342  /* Get a temporary random number for use in the residuals. Failout if retry fails */
343  if (startlen > 0)
344  {
345 #ifdef _IS64BIT
346  if ( (success= rdrand_64((uint64_t *) &temprand, 1)) != RDRAND_SUCCESS) return success;
347 #else
348  if ( (success= rdrand_32((uint32_t *) &temprand, 1)) != RDRAND_SUCCESS) return success;
349 #endif
350  }
351 
352  /* populate the starting misaligned block */
353  for (i = 0; i<startlen; i++)
354  {
355  start[i] = (unsigned char)(temprand & 0xff);
356  temprand = temprand >> 8;
357  }
358 
359  /* populate the central aligned block. Fail out if retry fails */
360 
361 #ifdef _IS64BIT
362  if ( (success= rdrand_get_n_64(length, (uint64_t *)(blockstart))) != RDRAND_SUCCESS) return success;
363 #else
364  if ( (success= rdrand_get_n_32(length, (uint32_t *)(blockstart))) != RDRAND_SUCCESS) return success;
365 #endif
366  /* populate the final misaligned block */
367  if (residual > 0)
368  {
369 #ifdef _IS64BIT
370  if ((success= rdrand_64((uint64_t *)&temprand, 1)) != RDRAND_SUCCESS) return success;
371 #else
372  if ((success= rdrand_32((uint32_t *)&temprand, 1)) != RDRAND_SUCCESS) return success;
373 #endif
374 
375  for (i = 0; i<residual; i++)
376  {
377  residualstart[i] = (unsigned char)(temprand & 0xff);
378  temprand = temprand >> 8;
379  }
380  }
381 
382  return RDRAND_SUCCESS;
383 }
